#!/bin/bash

        #Function to delete user if it exists#
function delete_user {

        userdel -r $username
        echo "User $username deleted successfully"
}

        #promt to Enter the usename manually#

read -p "Enter username to be deleted: " username


        #Loop created so that username should not be empty#

while [ -z "$username" ]; do

        echo "Username cannot be empty!!!"
        read -p "Enter username again: " username
done

function repeat {

        if grep -q "^$username:" /etc/passwd; then

                delete_user
        else
                echo "User $username does not exist."
        fi

}

repeat

	#Loop created if user to delete multiple user

read -p "Do you want to continue (Type Y or N): " choice

while [ "$choice" == Y ]; do

                read -p "Enter the username to be deleted: " username

                repeat

                read -p "Do you want to continue (Type Y or N): " choice

        if [ "$choice" == N ]; then

                echo ".................#Stopping the script#..............."

                break
        fi

done
